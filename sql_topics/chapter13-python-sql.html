✅ Context Manager — The Right Way (with)
💡
Using with ensures the connection is always closed — even if an error occurs. This prevents connection leaks in production. Always prefer this pattern.
Python
context_manager.py
import mysql.connector
from contextlib import contextmanager

# ── Reusable connection context manager ────────────────────
@contextmanager
def get_connection():
    conn = mysql.connector.connect(
        host="localhost", user="guvi_user",
        password="SecurePass@2024", database="guvi_db"
    )
    try:
        yield conn          # hand connection to the caller
    finally:
        conn.close()        # always runs — even on exception!

# ── Usage ──────────────────────────────────────────────────
with get_connection() as conn:
    cursor = conn.cursor()
    cursor.execute("SELECT COUNT(*) FROM students")
    print(cursor.fetchone()[0])
# connection automatically closed here ✅

🖱️ Cursor & execute()

The cursor is the object that sends SQL to the database and retrieves results. Think of it as the "pen" that writes SQL on the database.

📖 Cursor Types — Dictionary vs Tuple
Python
cursor_types.py
# Default cursor — returns tuples
cursor = conn.cursor()
cursor.execute("SELECT student_id, name, score FROM guvi_learners LIMIT 2")
row = cursor.fetchone()
print(row)           # ('S01', 'Priya Sharma', 85)
print(row[1])        # 'Priya Sharma'  — access by index

# Dictionary cursor — returns dicts (more readable)
cursor = conn.cursor(dictionary=True)   # mysql-connector
# cursor = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)  # psycopg2
cursor.execute("SELECT student_id, name, score FROM guvi_learners LIMIT 2")
row = cursor.fetchone()
print(row)           # {'student_id': 'S01', 'name': 'Priya Sharma', 'score': 85}
print(row['name'])  # 'Priya Sharma'  — access by key ✅ much clearer

Python

cursor.execute("""
    INSERT INTO students (name, email, city)

print(f"Inserted row ID: {cursor.lastrowid}")

    UPDATE guvi_learners
    SET score = 92 WHERE student_id = 'S01'
""")
conn.commit()

# DELETE
cursor.execute("DELETE FROM temp_table WHERE created_at < '2024-01-01'")
conn.commit()
print(f"Rows deleted: {cursor.rowcount}")
⚠️
Never forget conn.commit()For INSERT / UPDATE / DELETE, changes are in a transaction and won't be saved until you call conn.commit(). If you forget it, changes disappear when the connection closes.

📥 fetch() Methods

After execute(), you need to retrieve the results. Python DB-API gives three options depending on how many rows you need.

fetchone()

Returns ONE row as a tuple/dict. Returns None when no more rows. Best for lookups expecting a single result.

fetchall()

Returns ALL remaining rows as a list. Loads everything into memory — avoid on huge result sets.

fetchmany(n)

Returns next N rows. Memory-efficient for large results — process in batches without loading all at once.

💡 All Three in Practice — GUVI Learner Query
Python
fetch_methods.py
import mysql.connector

conn   = mysql.connector.connect(host="localhost", database="guvi_db",
                                   user="guvi_user", password="pass")
cursor = conn.cursor(dictionary=True)
cursor.execute("SELECT name, batch_id, score FROM guvi_learners ORDER BY score DESC")

# ── fetchone() — grab just the top scorer ─────────────────
top = cursor.fetchone()
print(f"Top scorer: {top['name']} — {top['score']}")
# Top scorer: Meera Iyer — 93

# ── fetchmany(3) — get next 3 rows (batch processing) ─────
next_three = cursor.fetchmany(3)
for row in next_three:
    print(f"  {row['name']:20s} {row['score']}")

# ── fetchall() — get ALL remaining rows ───────────────────
rest = cursor.fetchall()
print(f"\nRemaining learners: {len(rest)}")

cursor.close()
conn.close()

🔄 Iterating with fetchmany() — Memory-Efficient Pattern
Python
batch_processing.py
# Process 50,000 rows without loading all into memory
cursor.execute("SELECT * FROM zomato_orders WHERE YEAR(order_date) = 2024")

BATCH_SIZE = 500
total_processed = 0

while True:
    rows = cursor.fetchmany(BATCH_SIZE)
    if not rows:
        break                    # no more rows
    # process this batch ...
    total_processed += len(rows)
    print(f"Processed {total_processed} rows...")

print(f"✅ Done. Total: {total_processed} rows")

🛡️ Parameterized Queries — SQL Injection Safety

SQL injection is the #1 database attack. Never build SQL strings with Python f-strings or + concatenation. Always use parameterized queries.

☠️ The Dangerous Way — NEVER Do This
🚨
This is SQL Injection-vulnerable code!If a user types ' OR '1'='1 as their name, the query becomes: WHERE name = '' OR '1'='1' — which returns ALL rows, bypassing authentication.
Python — ❌ UNSAFE
UNSAFE — never do this!
# ❌ WRONG — string formatting opens SQL injection hole
user_input = "' OR '1'='1"   # attacker input

query = f"SELECT * FROM students WHERE name = '{user_input}'"
# Becomes: SELECT * FROM students WHERE name = '' OR '1'='1'
# This returns EVERY row in the table — data breach!

cursor.execute(query)   # ☠️ catastrophic
✅ The Safe Way — Parameterized Queries
Python — ✅ SAFE
safe_queries.py
# ✅ CORRECT — use %s placeholders (MySQL) or %s/%s (psycopg2)
# The driver handles escaping — injection is impossible

# Single parameter
cursor.execute(
    "SELECT * FROM students WHERE name = %s",
    (user_input,)          # tuple! trailing comma for single value
)

# Multiple parameters
cursor.execute(
    "SELECT name, score FROM guvi_learners WHERE batch_id = %s AND score > %s",
    ('B01', 75)
)

# INSERT with parameters — always use this pattern
cursor.execute(
    "INSERT INTO students (name, email, city) VALUES (%s, %s, %s)",
    ('Kavya Nair', 'kavya@guvi.in', 'Kochi')
)
conn.commit()
✅
Why %s is safeThe DB driver sends the query and parameters separately to the database. The DB engine knows they are data — not executable SQL. The attacker's ' OR '1'='1 is treated as a literal string, not SQL syntax.
🔍 Named Parameters — More Readable for Complex Queries
Python
named_params.py
# MySQL — use %(name)s syntax
cursor.execute(
    """SELECT name, score FROM guvi_learners
       WHERE batch_id = %(batch)s AND score > %(min_score)s""",
    {'batch': 'B01', 'min_score': 75}
)

# PostgreSQL — use %(name)s or :name (with SQLAlchemy)
cursor.execute(
    "SELECT name FROM students WHERE city = %(city)s",
    {'city': 'Chennai'}
)

⚡ Bulk Inserts & df.to_sql()

Writing data back to the database is just as important as reading. Bulk operations are dramatically faster than row-by-row inserts.

📤 df.to_sql() — DataFrame → Database Table
Python
to_sql.py
import pandas as pd
from sqlalchemy import create_engine

engine = create_engine("mysql+mysqlconnector://user:pass@localhost/guvi_db")

# Create a DataFrame
df_new = pd.DataFrame({
    'name'    : ['Kavya Nair', 'Dev Sharma', 'Ananya Rao'],
    'email'   : ['kavya@guvi.in', 'dev@guvi.in', 'ananya@guvi.in'],
    'city'    : ['Kochi', 'Jaipur', 'Hyderabad'],
    'batch_id': ['B02', 'B03', 'B01']
})

# Write to DB
df_new.to_sql(
    name      = 'students',    # table name
    con       = engine,
    if_exists = 'append',     # 'replace' | 'append' | 'fail'
    index     = False          # don't write DataFrame index as column
)
print(f"✅ {len(df_new)} rows inserted")  # ✅ 3 rows inserted
📋
if_exists parameter
'fail' → error if table exists (default)
'replace' → drops and recreates table — careful!
'append' → adds rows to existing table ← most common

def get_monthly_summary(month_str):
    y, m = month_str.split('-')
    df = pd.read_sql(text("""
        SELECT
          b.batch_name,
          c.course_name,
          COUNT(s.student_id)             AS total_learners,
          COUNT(s.quiz_score)             AS attempted,
          ROUND(AVG(s.quiz_score), 1)     AS avg_score,
          MAX(s.quiz_score)               AS top_score,
          MIN(s.quiz_score)               AS low_score,
          SUM(CASE WHEN s.quiz_score >= 75 THEN 1 ELSE 0 END) AS passed
        FROM   students s
        JOIN   batches  b ON s.batch_id  = b.batch_id
        JOIN   courses  c ON b.course_id = c.course_id
        WHERE  YEAR(s.enrolled_on)  = :yr
          AND  MONTH(s.enrolled_on) = :mo
        GROUP  BY b.batch_name, c.course_name
        ORDER  BY avg_score DESC
    """), engine, params={'yr': y, 'mo': m})
    df['pass_rate_%'] = (df['passed'] / df['attempted'] * 100).round(1)
    return df

df_summary = get_monthly_summary(MONTH)
print(df_summary.to_string(index=False))

def get_city_stats():
    return pd.read_sql("""
        SELECT
          city,
          COUNT(*) AS learners,
          ROUND(AVG(quiz_score), 1) AS avg_score,
          SUM(CASE WHEN DATEDIFF(CURDATE(), last_login) <= 7
                   THEN 1 ELSE 0 END) AS active_this_week
        FROM   students
        WHERE  quiz_score IS NOT NULL
        GROUP  BY city
        ORDER  BY learners DESC
        LIMIT  10
    """, engine)

def get_top_learners(n=10):
    return pd.read_sql(f"""
        SELECT name, city, batch_id, quiz_score AS score,
               DATEDIFF(CURDATE(), last_login) AS days_since_login
        FROM   students
        WHERE  quiz_score IS NOT NULL
        ORDER  BY quiz_score DESC LIMIT {n}
    """, engine)

df_city = get_city_stats()
df_top  = get_top_learners()

def generate_charts(df_summary, df_city, output_dir='.'):
    fig, axes = plt.subplots(1, 3, figsize=(18, 5))

    # Chart 1: Avg score per batch
    axes[0].bar(df_summary['batch_name'], df_summary['avg_score'],
               color='#3b82f6', edgecolor='white')
    axes[0].axhline(75, color='red', ls='--', label='Pass mark')
    axes[0].set_title(f'Avg Score by Batch — {MONTH}')
    axes[0].set_ylabel('Score'); axes[0].tick_params(axis='x', rotation=30)
    axes[0].legend()

    # Chart 2: Pass rate per batch
    colors = ['#16a34a' if x >= 80 else '#d97706'
              for x in df_summary['pass_rate_%']]
    axes[1].bar(df_summary['batch_name'], df_summary['pass_rate_%'], color=colors)
    axes[1].set_title('Pass Rate % by Batch')
    axes[1].set_ylim(0, 100); axes[1].tick_params(axis='x', rotation=30)

    # Chart 3: City distribution
    axes[2].barh(df_city['city'], df_city['learners'], color='#10b981')
    axes[2].set_title('Learners by City'); axes[2].set_xlabel('Count')

    plt.tight_layout()
    chart_path = os.path.join(output_dir, f'guvi_charts_{MONTH}.png')
    plt.savefig(chart_path, dpi=150, bbox_inches='tight')
    plt.close()
    print(f"✅ Charts saved: {chart_path}")
    return chart_path

chart_file = generate_charts(df_summary, df_city)

def export_excel(df_summary, df_city, df_top, chart_path):
    filename = f"GUVI_Report_{MONTH}.xlsx"

    with pd.ExcelWriter(filename, engine='openpyxl') as writer:

        # Sheet 1: Monthly summary
        df_summary.to_excel(writer, sheet_name='Batch Summary', index=False)

        # Sheet 2: City stats
        df_city.to_excel(writer, sheet_name='City Stats', index=False)

        # Sheet 3: Top learners
        df_top.to_excel(writer, sheet_name='Top Learners', index=False)

        # Sheet 4: Embed chart image
        ws_chart = writer.book.create_sheet('Charts')
        from openpyxl.drawing.image import Image as XLImage
        img = XLImage(chart_path)
        ws_chart.add_image(img, 'A1')

        # Style the header rows
        from openpyxl.styles import Font, PatternFill
        header_fill = PatternFill(fill_type='solid', fgColor='1a2744')
        header_font = Font(color='FFFFFF', bold=True)
        for sheet_name in ['Batch Summary', 'City Stats', 'Top Learners']:
            ws = writer.sheets[sheet_name]
            for cell in ws[1]:
                cell.fill = header_fill; cell.font = header_font

    print(f"✅ Report saved: {filename}")
    return filename
